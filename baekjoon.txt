#1463 1로 만들기   ###틀렸음
n = int(input())
time = 0

while n != 1:
    if n%3==0:
        n = n/3
        time +=1
    elif n%2==0:
        n = n/2
        time +=1
    else:
        n = n-1
        time +=1
    if n ==1:
        print(time)
        break
